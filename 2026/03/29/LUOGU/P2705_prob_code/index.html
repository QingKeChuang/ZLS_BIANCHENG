P2705 小球

题目描述

有 个红色盒子和 个蓝色盒子，还有 个红色小球和 个蓝色小球。每个盒子只能装一个小球，每个小球都要放在一个盒子里。

示例代码

#include <iostream>  
using namespace std; 
int main() {
    /**
     * 变量定义：
     * r (Red): 红球数量， b (Blue): 蓝球数量
     * c: 一个红球放进红盒的分数
     * d: 一个蓝球放进蓝盒的分数
     * e: “一对”异色球（1红+1蓝）放进杂色盒的单球分（总分为 2*e）
     * sum: 最终的最大总分
     * k: 数量比较标志位
     */
    int r, b, c, d, e, sum, k;
    // 使用 cin 输入数据，会自动根据空格区分不同的变量
    // 输入顺序：红球数 蓝球数 红盒分 蓝盒分 杂色盒分
    if (!(cin >> r >> b >> c >> d >> e)) return 0;
    // 【第一步：判断哪种球的数量更多】
    if (r > b) {
        k = 1; // 红球多于蓝球
    } else {
        k = 0; // 蓝球多于或等于红球
    }
    /**
     * 【第二步：贪心策略决策】
     * 我们需要对比两种装盒方式的收益：
     * 方式1：同色装（1红进红盒 + 1蓝进蓝盒）的总收益 = c + d
     * 方式2：异色装（1红1蓝进杂色盒）的总收益 = e + e (即 2*e)
     */
    if (c + d > e + e) {
        // 如果同色装更划算，则所有球各归各位
        sum = r * c + b * d;
    } 
    else {
        // 如果异色装（2*e）收益更高，我们要尽可能多地凑成“红蓝对”
        if (k == 1) {
            /** * 情况1：红球多 (r > b)
             * 能凑成的对数取决于数量少的球（蓝球），即 b 对。
             * 1. 凑成对的部分收益：b * 2 * e
             * 2. 剩下的红球只能单独放红盒：(r - b) * c
             */
            sum = b * 2 * e + (r - b) * c;
        } 
        else {
            /** * 情况2：蓝球多或相等 (b >= r)
             * 能凑成的对数取决于红球数量，即 r 对。
             * 1. 凑成对的部分收益：r * 2 * e
             * 2. 剩下的蓝球只能单独放蓝盒：(b - r) * d
             */
            sum = r * 2 * e + (b - r) * d;
        }
    }
    // 使用 cout 输出结果
    cout << sum << endl;
    return 0;
}```
